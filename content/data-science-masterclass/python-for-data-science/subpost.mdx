# Python for Data Science

Python has become the de facto standard for data science due to its simplicity, powerful libraries, and active community. This comprehensive guide will teach you everything you need to know about using Python for data science projects.

## Why Python for Data Science?

### Advantages of Python
- **Readable Syntax**: Easy to learn and understand

- **Versatility**: Works for web development, automation, and more

**Python vs R:**
- Python: General-purpose, better for production
- R: Specialized for statistics, excellent for research

**Python vs SQL:**
- Python: Complex data manipulation and analysis

- Python: Automation, large datasets, reproducibility


## Performance Optimization

### Efficient Data Operations

```python
# Use vectorized operations instead of loops
# Slow: Loop approach
result_slow = []
for value in df['Salary']:
    result_slow.append(value * 1.1)

# Fast: Vectorized approach
result_fast = df['Salary'] * 1.1

# Use .loc and .iloc for indexing
# Slow: Chain indexing
value_slow = df[df['Name'] == 'Alice']['Salary']

# Fast: Direct indexing
value_fast = df.loc[df['Name'] == 'Alice', 'Salary']

# Use categorical data for repeated strings
df['City_cat'] = df['City'].astype('category')

# Use appropriate data types
df['Age'] = df['Age'].astype('int8')  # If values are small
```

### Memory Optimization

```python
# Check memory usage
print(df.memory_usage(deep=True))

# Optimize data types
def optimize_dtypes(df):
    for col in df.columns:
        if df[col].dtype == 'object':
            df[col] = df[col].astype('category')
        elif df[col].dtype == 'int64':
            df[col] = pd.to_numeric(df[col], downcast='integer')
        elif df[col].dtype == 'float64':
            df[col] = pd.to_numeric(df[col], downcast='float')
    return df

df_optimized = optimize_dtypes(df.copy())
```

## Real-World Example: Sales Data Analysis

Let's analyze a complete sales dataset:

```python
# Generate sample sales data
np.random.seed(42)
n_records = 10000

sales_data = pd.DataFrame({
    'Date': pd.date_range('2023-01-01', periods=n_records, freq='H'),
    'Product': np.random.choice(['A', 'B', 'C', 'D'], n_records),
    'Region': np.random.choice(['North', 'South', 'East', 'West'], n_records),
    'Sales_Amount': np.random.normal(100, 30, n_records),
    'Quantity': np.random.poisson(5, n_records),
    'Customer_Type': np.random.choice(['New', 'Returning'], n_records)
})

# Data cleaning
sales_data['Sales_Amount'] = sales_data['Sales_Amount'].clip(lower=0)
sales_data['Revenue'] = sales_data['Sales_Amount'] * sales_data['Quantity']

# Exploratory Data Analysis
print("Dataset Overview:")
print(sales_data.info())
print("\nSummary Statistics:")
print(sales_data.describe())

# Time-based analysis
sales_data.set_index('Date', inplace=True)
daily_revenue = sales_data.resample('D')['Revenue'].sum()

# Product performance
product_performance = sales_data.groupby('Product').agg({
    'Revenue': ['sum', 'mean', 'count'],
    'Quantity': 'sum'
}).round(2)

# Regional analysis
regional_analysis = sales_data.groupby(['Region', 'Product'])['Revenue'].sum().unstack()

# Customer type analysis
customer_analysis = sales_data.groupby('Customer_Type')['Revenue'].agg([
    'sum', 'mean', 'count'
])

# Visualizations
fig, axes = plt.subplots(2, 2, figsize=(15, 10))

# Daily revenue trend
axes[0, 0].plot(daily_revenue.index, daily_revenue.values)
axes[0, 0].set_title('Daily Revenue Trend')
axes[0, 0].set_xlabel('Date')
axes[0, 0].set_ylabel('Revenue')

# Product revenue distribution
product_revenue = sales_data.groupby('Product')['Revenue'].sum()
axes[0, 1].bar(product_revenue.index, product_revenue.values)
axes[0, 1].set_title('Revenue by Product')
axes[0, 1].set_xlabel('Product')
axes[0, 1].set_ylabel('Total Revenue')

# Regional heatmap
sns.heatmap(regional_analysis, annot=True, fmt='.0f', ax=axes[1, 0])
axes[1, 0].set_title('Revenue Heatmap: Region vs Product')

# Sales amount distribution
axes[1, 1].hist(sales_data['Sales_Amount'], bins=30, edgecolor='black')
axes[1, 1].set_title('Sales Amount Distribution')
axes[1, 1].set_xlabel('Sales Amount')
axes[1, 1].set_ylabel('Frequency')

plt.tight_layout()
plt.show()

# Advanced insights
print("\n=== KEY INSIGHTS ===")
print(f"Total Revenue: ${sales_data['Revenue'].sum():,.2f}")
print(f"Average Daily Revenue: ${daily_revenue.mean():,.2f}")
print(f"Best Performing Product: {product_revenue.idxmax()}")
print(f"Best Performing Region: {sales_data.groupby('Region')['Revenue'].sum().idxmax()}")

# Correlation analysis
correlation_matrix = sales_data[['Sales_Amount', 'Quantity', 'Revenue']].corr()
print("\nCorrelation Matrix:")
print(correlation_matrix)
```

## Best Practices for Data Science Python

### Code Organization
```python
# Use functions for reusable code
def clean_sales_data(df):
    """Clean and preprocess sales data"""
    df = df.copy()
    df['Sales_Amount'] = df['Sales_Amount'].clip(lower=0)
    df['Revenue'] = df['Sales_Amount'] * df['Quantity']
    return df

def calculate_metrics(df):
    """Calculate key business metrics"""
    metrics = {
        'total_revenue': df['Revenue'].sum(),
        'avg_order_value': df['Revenue'].mean(),
        'total_orders': len(df)
    }
    return metrics

# Use classes for complex operations
class SalesAnalyzer:
    def __init__(self, data):
        self.data = data
        self.metrics = {}
    
    def calculate_trends(self):
        """Calculate time-based trends"""
        self.data.set_index('Date', inplace=True)
        self.daily_trends = self.data.resample('D')['Revenue'].sum()
        return self.daily_trends
    
    def segment_customers(self):
        """Segment customers based on behavior"""
        customer_segments = self.data.groupby('Customer_Type').agg({
            'Revenue': ['sum', 'mean', 'count']
        })
        return customer_segments
```

### Error Handling
```python
def safe_data_load(file_path):
    """Safely load data with error handling"""
    try:
        if file_path.endswith('.csv'):
            df = pd.read_csv(file_path)
        elif file_path.endswith('.xlsx'):
            df = pd.read_excel(file_path)
        else:
            raise ValueError("Unsupported file format")
        
        print(f"Successfully loaded {len(df)} rows")
        return df
    
    except FileNotFoundError:
        print(f"File {file_path} not found")
        return None
    except Exception as e:
        print(f"Error loading file: {str(e)}")
        return None

# Usage
df = safe_data_load('sales_data.csv')
if df is not None:
    # Proceed with analysis
    pass
```
